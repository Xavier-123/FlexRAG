  --embedding-base-url "http://127.0.0.1:19002/v1/embeddings" \
  --embedding-model "Qwen3-Embedding-0.6B" \
  --embedding-api-key "sk-1234567890" \
  --reranker-base-url "http://127.0.0.1:19003/v1/rerank" \
  --reranker-model "Qwen3-Reranker-0.6B" \

  --llm-base-url "https://api-inference.modelscope.cn/v1" \
  --llm-model "Qwen/Qwen3.5-35B-A3B" \
  --llm-api-key "ms-c429b084-79ba-4a00-a749-aae8681e902d" \
  --knowledge-persist-dir "./data/knowledge_persist_dir/hotpotqa/" \
  --top-k-retrieval 10 \
  --top-k-rerank 10 \
  --context-max-tokens 16384 \
  --max-concurrent-tasks 1 \
  --input-file "./data/hotpotqa-100.json" \
